1Buat Google Sheet Baru

Buka sheets.google.com → buat spreadsheet baru → beri nama "Laporan Kerja Harian".
Rename Sheet1 (untuk Electrical) dan tambahkan Sheet2 (untuk Mechanical). Header akan dibuat otomatis.

2Buka Apps Script & Paste Kode

Di Google Sheet → klik menu Extensions → Apps Script → hapus semua kode → paste kode berikut:

// ================================================ // LAPORAN KERJA HARIAN - Apps Script v5 (Electrical + Mechanical) // PASTE SELURUH KODE INI, LALU DEPLOY NEW VERSION // ================================================ var COLS = [ 'ID','Tanggal','Nama','Lokasi','Pekerjaan', 'Jam Mulai','Jam Selesai','Durasi', 'Material','QTY','UOM','Status','Remarks','Timestamp' ]; // Sheet 1 = Electrical, Sheet 2 = Mechanical function getSheet(dept) { var ss = SpreadsheetApp.getActive(); if (dept === 'mechanical') { // Pastikan Sheet2 ada, buat jika belum ada var sh = ss.getSheets()[1]; if (!sh) { sh = ss.insertSheet('Mechanical', 1); } return sh; } return ss.getSheets()[0]; // default: Sheet1 = Electrical } function ensureHeader(sheet) { if (sheet.getLastRow() === 0) { sheet.appendRow(COLS); sheet.getRange(1,1,1,COLS.length).setFontWeight('bold'); sheet.setFrozenRows(1); } } // ── doGet: getAll, edit, delete, test ────────────── function doGet(e) { try { var p = e.parameter; var action = p.action || 'getAll'; var dept = p.dept || 'electrical'; var sheet = getSheet(dept); if (action === 'test') { return json({status:'ok', msg:'Koneksi berhasil! Apps Script v5 aktif. Dept: ' + dept}); } if (action === 'getAll') { ensureHeader(sheet); var raw = sheet.getDataRange().getValues(); if (raw.length <= 1) return json({status:'ok', data:[]}); var headers = raw[0]; var rows = raw.slice(1).map(function(row, i) { var obj = {_row: i + 2}; headers.forEach(function(h, j) { obj[h] = row[j]; }); return obj; }); return json({status:'ok', data: rows}); } if (action === 'edit') { var rNum = parseInt(p._row); if (isNaN(rNum) || rNum < 2) { return json({status:'error', msg:'_row tidak valid: ' + p._row}); } ensureHeader(sheet); var lastTs = ''; try { lastTs = sheet.getRange(rNum, COLS.length).getValue(); } catch(x){} var vals = [[ p.fID || '', p.fTgl || '', p.fNama || '', p.fLok || '', p.fPek || '', p.fJm || '', p.fJs || '', p.fDur || '', p.fMat || '', p.fQty || '', p.fUom || '', p.fSts || '', p.fRem || '', lastTs || Utilities.formatDate(new Date(),'Asia/Jakarta','dd/MM/yyyy HH:mm:ss') ]]; sheet.getRange(rNum, 1, 1, COLS.length).setValues(vals); return json({status:'ok', updated: rNum}); } if (action === 'delete') { var rNum = parseInt(p._row); if (isNaN(rNum) || rNum < 2) { return json({status:'error', msg:'_row tidak valid: ' + p._row}); } sheet.deleteRow(rNum); return json({status:'ok', deleted: rNum}); } return json({status:'error', msg:'Action tidak dikenal: ' + action}); } catch(err) { return json({status:'error', msg: 'doGet error: ' + err.toString()}); } } // ── doPost: add ──────────────────────────────────── function doPost(e) { try { var body = JSON.parse(e.postData.contents); var dept = body.dept || 'electrical'; var sheet = getSheet(dept); ensureHeader(sheet); if (body.action === 'add') { var prefix = dept === 'mechanical' ? 'MC-' : 'EL-'; var id = prefix + Date.now(); body.rows.forEach(function(r) { sheet.appendRow([ id, r.tanggal, r.nama, r.lokasi, r.pekerjaan, r.jamMulai, r.jamSelesai, r.durasi, r.material, r.qty, r.uom, r.status, r.remarks, Utilities.formatDate(new Date(),'Asia/Jakarta','dd/MM/yyyy HH:mm:ss') ]); }); return json({status:'ok', id: id}); } return json({status:'error', msg:'Action POST tidak dikenal: ' + body.action}); } catch(err) { return json({status:'error', msg: 'doPost error: ' + err.toString()}); } } function json(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }

3Deploy Sebagai Web App

Klik Deploy → New deployment → Type: Web app
Execute as: Me | Who has access: Anyone → Klik Deploy → Izinkan akses → Copy URL yang muncul.
